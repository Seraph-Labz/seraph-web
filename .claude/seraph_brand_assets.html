# Using puppeteer-cli (recommended)
npx puppeteer-cli screenshot seraph_og_card.html seraph_og_card.png \
  --viewport 1200,630 --deviceScaleFactor 2

# Or with Playwright
npx playwright screenshot seraph_og_card.html seraph_og_card.png \
  --viewport-size=1200,630 --device-scale-factor=2
